Getting Input Values

Access user input with the .value property:

<input type="text" id="nameInput">
<button id="submitBtn">Submit</button>

<script>
let btn = document.getElementById("submitBtn");
let input = document.getElementById("nameInput");

btn.addEventListener("click", function() {
  let name = input.value;
  console.log("User entered: " + name);
});
</script>

Real-Time Input (input event)

Update content as the user types:

<input type="text" id="liveInput">
<p>You typed: <span id="output"></span></p>

<script>
let input = document.getElementById("liveInput");
let output = document.getElementById("output");

input.addEventListener("input", function() {
  output.textContent = input.value;
});
</script>

Building a Counter

<p>Count: <span id="counter">0</span></p>
<button id="increaseBtn">+</button>
<button id="decreaseBtn">-</button>

<script>
let count = 0;
let display = document.getElementById("counter");

document.getElementById("increaseBtn").addEventListener("click", function() {
  count++;
  display.textContent = count;
});

document.getElementById("decreaseBtn").addEventListener("click", function() {
  count--;
  display.textContent = count;
});
</script>

Input Event vs Change Event

  • input: Fires every time the value changes (as you type)
  • change: Fires when focus leaves the field

For live preview: use input

For form submission: use change or submit

Live Calculator Example

<input type="number" id="num1" placeholder="Number 1">
<input type="number" id="num2" placeholder="Number 2">
<p>Sum: <span id="sum">0</span></p>

<script>
let num1 = document.getElementById("num1");
let num2 = document.getElementById("num2");
let sum = document.getElementById("sum");

function calculate() {
  let n1 = parseFloat(num1.value) || 0;
  let n2 = parseFloat(num2.value) || 0;
  sum.textContent = n1 + n2;
}

num1.addEventListener("input", calculate);
num2.addEventListener("input", calculate);
</script>

Validate User Input

function validateEmail(email) {
  if (email.includes("@") && email.includes(".")) {
    return true;
  } else {
    return false;
  }
}

let emailInput = document.getElementById("email");
emailInput.addEventListener("change", function() {
  if (validateEmail(emailInput.value)) {
    console.log("Valid email!");
  } else {
    console.log("Invalid email!");
  }
});
